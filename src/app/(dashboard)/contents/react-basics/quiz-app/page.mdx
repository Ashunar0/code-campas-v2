```jsx
import { useState } from "react";
import { quizData } from "./data/quizData";
import "./App.css";

function App() {
  const [currentIndex, setCurrentIndex] = useState(0);
  const [score, setScore] = useState(0);
  const [showResult, setShowResult] = useState(false);
  const [selectedAnswer, setSelectedAnswer] = useState(null);
  const [isAnswered, setIsAnswered] = useState(false);

  const currentQuiz = quizData[currentIndex];

  const handleAnswer = (choiceIndex) => {
    if (isAnswered) return;

    setSelectedAnswer(choiceIndex);
    setIsAnswered(true);

    if (choiceIndex === currentQuiz.correctIndex) {
      setScore(score + 1);
    }
  };

  const handleNext = () => {
    if (currentIndex < quizData.length - 1) {
      setCurrentIndex(currentIndex + 1);
      setSelectedAnswer(null);
      setIsAnswered(false);
    } else {
      setShowResult(true);
    }
  };

  const handleRestart = () => {
    setCurrentIndex(0);
    setScore(0);
    setShowResult(false);
    setSelectedAnswer(null);
    setIsAnswered(false);
  };

  // çµæœç”»é¢
  if (showResult) {
    return (
      <div className="app">
        <h1>ã‚¯ã‚¤ã‚ºçµ‚äº†ï¼</h1>
        <p className="result">
          ã‚ãªãŸã®ã‚¹ã‚³ã‚¢: {score} / {quizData.length}
        </p>
        <button onClick={handleRestart}>ã‚‚ã†ä¸€åº¦</button>
      </div>
    );
  }

  // ã‚¯ã‚¤ã‚ºç”»é¢
  return (
    <div className="app">
      <div className="progress">
        å•é¡Œ {currentIndex + 1} / {quizData.length}
      </div>

      <h2 className="question">{currentQuiz.question}</h2>

      <div className="choices">
        {currentQuiz.choices.map((choice, index) => {
          let className = "choice-btn";
          if (isAnswered) {
            if (index === currentQuiz.correctIndex) {
              className += " correct";
            } else if (index === selectedAnswer) {
              className += " wrong";
            }
          }

          return (
            <button
              key={index}
              className={className}
              onClick={() => handleAnswer(index)}
              disabled={isAnswered}
            >
              {choice}
            </button>
          );
        })}
      </div>

      {isAnswered && (
        <div className="feedback">
          <p>
            {selectedAnswer === currentQuiz.correctIndex
              ? "æ­£è§£ï¼ğŸ‰"
              : "ä¸æ­£è§£...ğŸ˜¢"}
          </p>
          <button onClick={handleNext}>
            {currentIndex < quizData.length - 1 ? "æ¬¡ã®å•é¡Œ" : "çµæœã‚’è¦‹ã‚‹"}
          </button>
        </div>
      )}

      <p className="score">ç¾åœ¨ã®ã‚¹ã‚³ã‚¢: {score}</p>
    </div>
  );
}

export default App;
```
